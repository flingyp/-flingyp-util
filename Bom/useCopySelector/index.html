Online Demo #

@flypeng-tool：useCopy function
复制结果：失败

useCopy
copy text to clipboard by function
<template>
  <div class="copyDom">@flypeng-tool：useCopy function</div>
  <button @click="copy">点击</button>
  <div>复制结果：{{ copyResult ? '成功' : '失败' }}</div>
</template>

<script lang="ts" setup>
  import { ref } from 'vue'
  import { useCopySelector } from '@flypeng/browser'

  const copyResult = ref(false)

  const copy = () => {
    copyResult.value = useCopySelector('.copyDom')
  }
</script>

  button {
    width: 80px;

</style>
